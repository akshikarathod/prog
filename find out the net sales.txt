// find out the net sales//
#include<stdio.h>
int main()
{
    float net_sales,gross_sales,discount;
    net_sales=gross_sales-discount;
    printf("Enter your sales:\n");
    scanf("%f",&gross_sales);
    if(gross_sales>20000)
    {
    discount=0.15*gross_sales;                       //15%//
    net_sales=gross_sales- discount;
    printf("your net sales= %.2f\n",net_sales);
    }
    else
    if(gross_sales>10000)
    {
    discount=0.10*gross_sales;                        //10%//
    net_sales=gross_sales-discount;
    printf("your net sales = %.2f\n",net_sales);
    }
    else
    {
    discount=0.05*gross_sales;                         //5%//
    printf("your net sales = %.2f\n",net_sales);
    }
}
