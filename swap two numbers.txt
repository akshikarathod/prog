// swap two  numbers//
#include <stdio.h>
int main() {
    int a, b;

    printf("Enter two numbers: \n");
    scanf("%d %d", &a, &b);

    printf("Before swapping: a = %d, b = %d\n", a, b);

    a = a + b;
    b = a - b;
    a = a - b;

    printf("After swapping: a = %d, b = %d\n", a, b);
    
}
output
Enter two numbers: 
8 4
Before swapping: a = 8, b = 4
After swapping: a = 4, b = 8
