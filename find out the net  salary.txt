#include<stdio.h>
int main()
{
    float gross_salary,net_salary,allowances,deductions;
    net_salary=gross_salary+allowances-deductions;
    printf("Enter your salary:\n");
    scanf("%f",&gross_salary);
    if(gross_salary>10000)
    {
     allowances= 0.10*gross_salary;            //10%//
     deductions=0.03*gross_salary;             //3%//
     net_salary=gross_salary+allowances-deductions;
     printf("your net salary = %.2f\n",net_salary);
    }
     else 
     if(gross_salary>5000)
     {
      allowances=0.07*gross_salary;             //7%//
      deductions=0.02*gross_salary;             //2%//
      net_salary=gross_salary+allowances-deductions;
      printf("your net salary = %.2f\n",net_salary);
     }
     else
     {
         allowances=0;                          //no extra allowances//
         deductions=0;                          //no deductions//
         net_salary=gross_salary+allowances-deductions;
         printf("your net salary = %.2f\n",net_salary);
     }
     
    }
