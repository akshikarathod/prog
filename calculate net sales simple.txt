//calculate net sales//
#include<stdio.h>
int main()
{
    float net_sales,gross_sales,discount;
    
    printf("enter your sales:");
    scanf("%f",&gross_sales);
    
    discount=0.10*gross_sales;
    net_sales=gross_sales-discount;
    
    printf("your net sales is = %.2f",net_sales);
}
